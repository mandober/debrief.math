1 = {∅}     One
𝔹 = {⊥, ⊤}

ℕ = {1,2,3,...}
ℤ = {...,-1,0,1,...}
ℚ = { p/q | p,q ∈ ℤ, q ≠ 0 }
ℝ
ℂ


ℕᐩ = ℕ \ {0}
ℕ₀ = ℕ ⋃ {0}
ℕ = ℤᐩ
ℙ = { p ∈ ℕ | ∀x ∈ ℕ. x ∉ {1,p} -> x ∤ p } = primes
ℤ⁻ = ℤ \ ℕ₀

𝕎 = ℕ ⋃ {0}
ℕᐩ = ℕ \ {0}
ℤ = ℤ⁻ ⋃ {0} ⋃ ℕᐩ

ℚ = ℤ ⋃ 𝔽
ℚ = fromList [ p % q | p <- [1..], q <- [1..] ] :: Set Integer


ℝ = ℚ ⋃ 𝕀
ℝ = 𝕋 ⋃ 𝔸
ℝ = ℝ⁻ ⋃ {0} ⋃ ℝᐩ
ℂ = ℝ ⋃ 𝕁

- subset-of relations are intuitively: ℕ ⊆ ℤ ⊆ ℚ ⊆ ℝ ⊆ ℂ
- however, unintuitively: ℕ = ℤ = ℚ < ℝ < ℂ
